otto_arm_res_events.1 = {
	#Armenian Resistance Start
	type = country_event
	placement = IMP
	title = otto_arm_res_events.1.t
	desc = otto_arm_res_events.1.d
	flavor = otto_arm_res_events.1.f

		video = "gfx/event_pictures/unspecific_signed_contract.bk2"
	}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/signed_contract"
	icon = "gfx/interface/icons/event_icons/unify.dds"
	duration = 3
	trigger = {
		year > 1875
		this = c:TUR
	}

		}
		add_journal_entry = {
			type = otf_armenian_integration_je
		}
	}
}

otto_arm_res_events.2 = {
	#Yerel Ermeni İsyanı
	type = country_event
	placement = IMP
	title = otto_arm_res_events.2.t
	desc = otto_arm_res_events.2.d
	flavor = otto_arm_res_events.2.f
	event_image = {
		video = "gfx/event_pictures/unspecific_signed_contract.bk2"
	}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/signed_contract"
	icon = "gfx/interface/icons/event_icons/unify.dds"
	duration = 3
	immediate = {
		random_scope_state = {
			limit = {
				owner = c:TUR
				state_region = {
					is_homeland = cu:armenian
				}
			}
			save_scope_as = armenian_rebel_state
		}
	}
	option = {
		#Tüm isyancıları öldür, sivilleri sürgün et
		name = otto_arm_res_events.2.a
		scope:armenian_rebel_state = {
			add_modifier = {
				name = otf_armenian_local_separatism
				months = normal_modifier_time
			}
			every_scope_pop = {
				change_pop_culture = {
					value = 0.2
					target = cu:turkish
				}
				change_pop_culture = {
					value = 0.15
					target = cu:kurdish
				}
			}
		}
		random_scope_state = {
			limit = {
				owner = c:TUR
				state_region = {
					is_homeland = cu:mashriqi
				}
			}
			every_scope_pop = {
				change_pop_culture = {
					value = 0.1
					target = cu:armenian
				}
			}
		}
		change_variable = {
			name = anti_armenian_policy_var
			add = 3
		}
	}
	option = {
		#İsyancıları öldür, esirleri sürgün et
		name = otto_arm_res_events.2.b
		scope:armenian_rebel_state = {
			add_modifier = {
				name = otf_armenian_local_separatism_2
				months = normal_modifier_time
			}
			every_scope_pop = {
				every_scope_pop = {
					change_pop_culture = {
						value = 0.1
						target = cu:turkish
					}
					change_pop_culture = {
						value = 0.05
						target = cu:kurdish
					}
				}
			}
		}
		random_scope_state = {
			limit = {
				owner = c:TUR
				state_region = {
					is_homeland = cu:mashriqi
				}
			}
			every_scope_pop = {
				change_pop_culture = {
					value = 0.05
					target = cu:armenian
				}
			}
		}
		change_variable = {
			name = anti_armenian_policy_var
			add = 2
		}
	}
	option = {
		#Kargaşıyı önle
		name = otto_arm_res_events.2.c
		add_modifier = {
			name = otf_prevent_separatism
			months = normal_modifier_time
		}
		change_variable = {
			name = armenian_integration_var
			add = 3
		}
	}
	option = {
		#Müdehale etme
		name = otto_arm_res_events.2.d
		scope:armenian_rebel_state = {
			add_modifier = {
				name = otf_armenian_local_separatism_2
				months = normal_modifier_time
			}
			every_scope_pop = {
				change_pop_culture = {
					value = 0.15
					target = cu:armenian
				}
			}
		}
		change_variable = {
			name = armenian_integration_var
			add = 2
		}
	}
}

otto_arm_res_events.5 = {
	#Hamidiye Alayları
	type = country_event
	placement = IMP
	title = otto_arm_res_events.5.t
	desc = otto_arm_res_events.5.d
	flavor = otto_arm_res_events.5.f
	event_image = {
		video = "gfx/event_pictures/unspecific_signed_contract.bk2"
	}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/signed_contract"
	icon = "gfx/interface/icons/event_icons/unify.dds"
	duration = 2
	trigger = {
		has_variable = armenian_resistance_start
		AND = {
			NOT = {
				has_variable = completed_armeninan_integration
			}
			NOT = {
				has_variable = completed_anti_armenian_policy
			}
		}
		NOT = {
			has_variable = hamidiye_event_triggered
		}
	}
	immediate = {
		set_variable = hamidiye_event_triggered
	}
	option = {
		#Hamidiye Alaylarını Kur
		name = otto_arm_res_events.5.a
		add_loyalists = {
			culture = cu:kurdish
			value = 0.2
		}
		every_state = {
			limit = {
				owner = c:TUR
				state_region = {
					is_homeland = cu:armenian
				}
			}
			add_modifier = {
				name = otf_hamidiye_barracks
				months = normal_modifier_time
			}
		}
		add_modifier = {
			name = otf_foundation_hamidiye
			months = normal_modifier_time
		}
		change_variable = {
			name = anti_armenian_policy_var
			add = 4
		}
		set_variable = hamidiye_alay_foundation
	}
	option = {
		#Böyle bir orduya ihtiyacımız yok
		name = otto_arm_res_events.5.b
		add_loyalists = {
			culture = cu:armenian
			value = 0.1
		}
		change_variable = {
			name = armenian_integration_var
			add = 2
		}
	}
}

otto_arm_res_events.6 = {
	#Kürt-Ermeni Çatışması
	type = country_event
	placement = IMP
	title = otto_arm_res_events.6.t
	desc = otto_arm_res_events.6.d
	flavor = otto_arm_res_events.6.f
	event_image = {
		video = "gfx/event_pictures/unspecific_signed_contract.bk2"
	}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/signed_contract"
	icon = "gfx/interface/icons/event_icons/unify.dds"
	duration = 1
	immediate = {
		random_scope_state = {
			limit = {
				owner = c:TUR
				state_region = {
					is_homeland = cu:armenian
					is_homeland = cu:kurdish
				}
			}
			save_scope_as = armenian_vs_kurdish_state
		}
	}
	option = {
		#Kürtleri Destekle
		name = otto_arm_res_events.6.a
		scope:armenian_vs_kurdish_state = {
			add_modifier = {
				name = otf_armenian_local_separatism_2
				months = normal_modifier_time
			}
			every_scope_pop = {
				change_pop_culture = {
					value = 0.2
					target = cu:kurdish
				}
			}
			add_radicals = {
				culture = cu:armenian
				value = 0.1
			}
		}
		if = {
			limit = {
				has_variable = hamidiye_alay_foundation
			}
			change_variable = {
				name = anti_armenian_policy_var
				add = 4
			}
		}
		else = {
			change_variable = {
				name = anti_armenian_policy_var
				add = 2
			}
		}
	}
	option = {
		#Ermenileri Destekle
		name = otto_arm_res_events.6.b
		scope:armenian_vs_kurdish_state = {
			add_modifier = {
				name = otf_armenian_local_separatism_2
				months = normal_modifier_time
			}
			every_scope_pop = {
				change_pop_culture = {
					value = 0.2
					target = cu:armenian
				}
			}
			add_radicals = {
				culture = cu:kurdish
				value = 0.1
			}
		}
		if = {
			limit = {
				has_variable = hamidiye_alay_foundation
			}
			change_variable = {
				name = armenian_integration_var
				add = 4
			}
		}
		else = {
			change_variable = {
				name = armenian_integration_var
				add = 2
			}
		}
	}
	option = {
		#Do nothing
		name = otto_arm_res_events.6.c
		scope:armenian_vs_kurdish_state = {
			add_radicals = {
				culture = cu:kurdish
				value = 0.15
			}
			add_radicals = {
				culture = cu:kurdish
				value = 0.15
			}
		}
	}
}

otf_armenian_resistance_je = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"
	complete = {
		NOT = {
			has_variable = armenian_bad_sol_var
		}
	}
	on_complete = {
		every_state = {
			limit = {
				owner = c:TUR
				state_region = {
					is_homeland = cu:armenian
				}
			}
			add_modifier = {
				name = otf_armenian_become_loyal
				months = -1
			}
		}
	}
	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					any_state = {
						owner = c:TUR
						state_region = {
							is_homeland = cu:armenian
						}
						any_scope_pop = {
							culture = cu:armenian
							standard_of_living <= 15
						}
					}
				}
				change_variable = {
					name = armenian_bad_sol_var
					add = 1
				}
			}
		}
	}
	current_value = {
		value = var:armenian_bad_sol_var
	}
	goal_add_value = {
		add = 100
	}
	fail = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}
	on_fail = {
		trigger_event = {
			id = otto_arm_res_events.1
		}
		set_variable = armenian_resistance_start
	}
	progressbar = yes
	weight = 10000
	should_be_pinned_by_default = yes
}

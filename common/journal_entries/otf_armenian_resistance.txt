otf_armenian_resistance = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"
	is_shown_when_inactive = {
		this = c:TUR
	}
	possible = {
		# year > 1875
	}
	complete = {
		year > 1885
		var:armenian_bad_sol_var < 3
	}
	on_complete = {
		#Ermenileri sadık yap
	}
	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					any_scope_state = {
						owner = c:TUR
						state_region = {
							is_homeland = cu:armenian
						}
						standard_of_living < 15
					}
				}
				change_variable = {
					name = armenian_bad_sol_var
					add = 1
				}
			}
		}
		random_events = {
			100 = 0
			10 = tanzimat_events.5
		}
	}
	current_value = {
		value = var:armenian_bad_sol_var
	}
	goal_add_value = {
		add = 3
	}
	fail = {
		var:armenian_bad_sol_var >= 3
	}
	invalid = {
		trigger_event = {
			id = otto_arm_res_events.1
		}
		set_variable = armenian_resistance_start
	}
	progressbar = yes
	weight = 10000
	should_be_pinned_by_default = yes
}

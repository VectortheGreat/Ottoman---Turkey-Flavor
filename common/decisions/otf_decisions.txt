otf_fate_of_egypt = {
	is_shown = {
		has_variable = fate_of_egypt_dcs
		NOT = {
			has_variable = clr_egypt_dsc
		}
		exists = c:EGY
	}
	possible = {
		this = c:TUR
		c:EGY = {
			is_subject_of = c:TUR
			relations:root >= relations_threshold:amicable
		}
	}
	when_taken = {
		trigger_event = {
			id = otto_egypt_events.2
		}
		set_variable = clr_egypt_dsc
	}
	ai_chance = {
		base = 0
		modifier = {
			trigger = {
				bureaucracy > 0
			}
			add = 5
		}
		modifier = {
			OR = {
				is_diplomatic_play_committed_participant = yes
				is_at_war = yes
			}
			add = -1000
		}
	}
}

otf_annex_islamic_subjects_dcs = {
	is_shown = {

			has_variable = i_am_islamic_subject_ok
		}
		NOT = {
			has_variable = can_annex_islamic_subjects
		}

	possible = {
		this = c:TUR
		prestige >= 10000
		any_country = {
			has_variable = i_am_islamic_subject_ok
		}
	}
	when_taken = {
		set_variable = can_annex_islamic_subjects
	}
	ai_chance = {
		base = 0
		modifier = {
			trigger = {
				bureaucracy > 0
			}
			add = 5
		}
		modifier = {
			OR = {
				is_diplomatic_play_committed_participant = yes
				is_at_war = yes
			}
			add = -1000
		}
	}
}
